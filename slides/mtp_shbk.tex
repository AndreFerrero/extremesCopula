\section{Extreme Value Theory: Foundations}

\begin{frame}{GEV: The Limit Distribution of Maxima}

  Assume $X_1, \dots, X_n$ are \alert{i.i.d.} random variables from a distribution $F$. Let $M_n = \max(X_1, \dots, X_n)$. \pause
  
	If there exist sequences $a_n > 0$ and $b_n$ such that:
	\[
	P\left( \frac{M_n - b_n}{a_n} \le x \right) \xrightarrow{n \to \infty} G_\xi(x)
	\]
	then $G_\xi$ is a \alert{Generalized Extreme Value (GEV)} distribution. We say that $F$ is in the \textbf{maximum domain of attraction} (MDA) of $G_\xi$. \pause

  The \textbf{shape parameter} $\xi$ (Extreme Value Index) defines the margin tail behavior:
  
  \begin{itemize}
    \item \textbf{$\xi > 0$:} \textbf{Fréchet} (Heavy-tailed, e.g., Pareto, Cauchy)
    \item \textbf{$\xi = 0$:} \textbf{Gumbel} (Exponential-like, e.g., Normal, Gamma)
    \item \textbf{$\xi < 0$:} \textbf{Weibull} (Bounded upper tail, e.g., Uniform, Beta)
  \end{itemize}
  
\end{frame}

\begin{frame}{GPD: The Limit Distribution of Exceedances}
    For a sufficiently high threshold $u$, we consider the distribution of the \alert{exceedances} $Y = X - u$, given that $X > u$. If $F \in MDA(G_\xi)$, with $G_\xi$ a GEV distribution, then:
    
    
    \[
    P(X - u \le y \mid X > u) \xrightarrow{u \to u_F} H_{\xi, \sigma}(y) = \begin{cases}
		1 - \left( 1 + \xi \frac{y}{\sigma} \right)^{-1/\xi}, & \xi \neq 0, y > 0 \\
		1 - \exp\left( -\frac{y}{\sigma} \right), & \xi = 0, y > 0
		\end{cases}
    \]
   
		where $H_{\xi, \sigma}$ is the \alert{Generalized Pareto Distribution (GPD)}

\end{frame}

\begin{frame}{Hill Estimator: a semi-parametric approach for $\xi > 0$}
    Assume $F$ has a regularly varying tail: $\bar{F}(x) = x^{-1/\xi} L(x)$, with $\xi > 0$ and $L(x)$ a slowly varying function. \pause

    Given the order statistics $X_{(1)} \le X_{(2)} \le \dots \le X_{(n)}$, the Hill estimator is defined as:
    \[
    \hat{\xi}_H(k) = \frac{1}{k} \sum_{i=1}^{k} \log X_{(n-i+1)} - \log X_{(n-k)}
    \]
		
\end{frame}

\begin{frame}{Hill Plot: Choosing the Number of Order Statistics}
	\begin{figure}
		\centering
		\includegraphics[width=\textwidth]{figures/hill_plot.png}
	\end{figure}
\end{frame}


 \begin{frame}{Asymptotic Normality of Hill Estimator}
    Under certain (second order) conditions on the tail of the distribution, Hill's estimator is asymptotically normal but biased:

        \[
        \sqrt{k} \left( \hat{\xi}_H - \xi \right) \xrightarrow{d} N\left( \frac{\lambda}{1-\rho}, \xi^2 \right)
        \] with $\rho \leq 0$ and $\lambda$ a finite constant.
        \pause

    \begin{itemize}
        \item \textbf{Bias Correction:} Requires estimating $\rho$.
        \item \textbf{Dependence:} \parencite{dehaan2016} shows correction under \alert{$\beta$-mixing}
    \end{itemize}
\end{frame}

\begin{frame}{Extended GPD: Avoiding Threshold Selection}
    The idea is to transform the GPD by a suitable distortion function $G(u)$ \parencite{naveau2016}. \pause

    \begin{block}{Power Transformation}
        For a random variable $X > 0$, the EGPD CDF is:
        \[
        F(x) = \left[ H_{\xi, \sigma}(x) \right]^\kappa
        \]
        where $H_{\xi, \sigma}(x)$ is the GPD CDF and \alert{$\kappa > 0$} is a bulk parameter.
    \end{block} \pause

    \textbf{Tail Consistency:} For any $\kappa > 0$, the tail behavior remains governed by $\xi$. The EGPD and GPD share the \alert{same extreme value index}.
\end{frame}

\begin{frame}{Mixing Conditions and Extremal Dependence}

\textbf{General idea of mixing conditions:}

For a sufficiently large lag $l$, distant blocks of a mixing time series behave approximately independently:
\[
(X_1, \dots, X_i)
\;\overset{\text{(approx. independence)}}{\longleftrightarrow}\;
(X_{i+l}, \dots, X_{i+l+j})
\] \pause

\textbf{Extension to extremes: $D(u_n)$ condition} \parencite{leadbetter1974}

The same idea is applied only to exceedances:
\[
(X_1 > u_n, \dots, X_i > u_n)
\;\overset{\text{(approx. independence)}}{\longleftrightarrow}\;
(X_{i+l} > u_n, \dots, X_{i+l+j} > u_n)
\]

\end{frame}

\begin{frame}{Distribution of Maxima of Stationary Sequences}

    Assume $\{X_n\} \sim F$ is a stationary sequence satifying $D(u_n)$ with maximum $M_n$. Assume also that for the i.i.d. sequence $\{X^*_n\} \sim F$, the associated maxima $M^*_n$ has GEV limit $G_\xi$. \pause Then, for a constant $0 < \gamma \leq 1$ (\alert{extremal index}),

    \[
    P \left(\frac{M_n - b^*_n}{a^*_n} \le x \right) \xrightarrow{n \to \infty} G_\xi^\gamma(x).
    \] \pause

    \textbf{Note:} Under i.i.d, $\gamma = 1$ and therefore we recover the standard result.
\end{frame}

\section{Copula Markov Chains}

\begin{frame}{Copulas: Decoupling Margins from Dependence}
    A \textbf{Copula} is a multivariate distribution function with uniform marginals on $[0,1]$. \pause

    \begin{block}{Sklar's Theorem (Bivariate Case)}
        For a joint distribution $F_{X,Y}$ with continuous marginals $F_X$ and $F_Y$, there exists a unique copula $C$ such that:
        \[ F_{X,Y}(x,y) = C(F_X(x), F_Y(y)) \] and viceversa.
    \end{block} \pause

    \textbf{Relevance to EVT:}
    \begin{itemize}
        \item We can choose a specialized margin (e.g., \alert{EGPD}) for extremes.
        \item We can choose a copula $C$ that captures \alert{tail dependence} (the probability of co-occurring extremes).
    \end{itemize}
\end{frame}

\begin{frame}{Copula Markov Chains}

    \begin{overprint}
        \onslide<1>
        A stationary (first-order) Markov chain $\{X_t\}$ can be constructed from a marginal distribution $F$ and a copula $C$ for the pair $(X_{t-1}, X_t)$.

        \begin{block}{Joint Distribution}
            \[ F_{X_{t-1}, X_t}(x_{t-1}, x_t) = C(F(x_{t-1}), F(x_t)) \]
        \end{block}

        \begin{block}{Conditional Distribution}
            \[ 
            P(X_t \le x_t \mid X_{t-1} = x_{t-1}) = \frac{\partial}{\partial u} C(u, F(x_t)) \Big|_{u=F(x_{t-1})} 
            \]
        \end{block}
				
		\onslide<2>
        \begin{block}{Properties}
            Copula Markov Chains generally satisfy \alert{mixing conditions} for many copula choices (e.g. Gumbel) \parencite{chen2009,longla2012}
        \end{block}

        \onslide<3>
        \begin{block}{Sampling (Rosenblatt Transform)}
            \vspace{0.2cm}
            \begin{enumerate}
                \item Generate $V_t \sim U(0,1)$
                \item Set $U_1 = V_1$ and $U_t = C^{-1}(V_t \mid U_{t-1})$ (Numerical Inversion)
                \item $X_t = F^{-1}(U_t)$
            \end{enumerate}
        \end{block}
    \end{overprint}
\end{frame}

\begin{frame}{Inference for Copula Models}
    \begin{overprint}
        \onslide<1>
        There are different ways to estimate copula models. For a review, see \parencite{choros2010}

        \onslide<2>
        \begin{block}{1. Inference Functions for Margins (IFM)}
		\vspace{0.2cm}
            A two-step estimator:
            \begin{enumerate}
                \item Estimate marginal parameters $\hat{\bm{\phi}}$ (or non-parametrically) assuming i.i.d. observations.
                \item Estimate copula parameters $\hat{\bm{\theta}}$ given the fitted margins by maximising copula log-likelihood.
            \end{enumerate}
        \end{block}

        \onslide<3>
        \begin{block}{2. Full Joint Likelihood}
					\vspace{0.2cm}
            Simultaneous estimation by maximizing the joint log-likelihood. For a copula markov model:
            \[ \ell((\bm{\phi},\bm{\theta})) = \sum_{t=1}^n \log f(x_t; \bm{\phi}) + \sum_{t=2}^n \log c(F(x_{t-1}; \bm{\phi}), F(x_t; \bm{\phi}); \bm{\theta}) \]
        \end{block}
    \end{overprint}
\end{frame}
